"""
This notebook solves the Multi-Armed Bandit problem using Bayesian updates on Thompson sampling of beta distributions. 
This focuses on finding which one is best quickly and exploiting it. Thompson sampling balances exploration/exploitation.
"""

import scipy.stats as ss
from numpy import *

class bandit_estimator_betadist(object):
    def __init__(self, num_candidates=2, prior=(1.0,1.0)):
        self.num_candidates = num_candidates
        self.prior = prior
        self.trials = zeros(shape=(num_candidates,), dtype=int)
        self.wins = zeros(shape=(num_candidates,), dtype=int)
        
    def update_on_trial(self, candidate_id, success):
        self.trials[candidate_id] += 1
        if (success):
            self.wins[candidate_id] += 1
            
    def pick_candidate_for_trial(self):
        samples = []
        for i in range(self.num_candidates):
            dist = ss.beta(self.prior[0]+self.wins[i],
                        self.prior[1]+self.trials[i]-self.wins[i])
            samples += [ dist.rvs() ]
        return samples.index( max(samples) )
    
    def conf_interval(self, conf, trials, wins):
        return ss.beta.interval(conf, self.prior[0]+wins, self.prior[1]+trials-wins)
    
    def candidate_mode(self, candidate_id):
        # mode is (a-1)/(a+b-2)
        # a = prior[0]+wins, b = prior[1]+trials-wins
        try:
            return ( (self.prior[0]+self.wins[candidate_id]-1) / (self.prior[0]+self.prior[1]+self.trials[candidate_id]-2) )
        except:
            return None        
    
    def mode(self, trials, wins):
        # mode is (a-1)/(a+b-2)
        # a = prior[0]+wins, b = prior[1]+trials-wins
        try:
            return ( (self.prior[0]+wins-1) / (self.prior[0]+self.prior[1]+trials-2) )
        except:
            return None
        
    def generate_samples(self, num_samples, num_trials, num_wins):
        dist = ss.beta(self.prior[0]+num_wins,
                       self.prior[1]+num_trials-num_wins)
        return dist.rvs(size=num_samples) 
    
    def evaluate_progress(self, num_samples):
        # generate samples of the candidate estimate distributions
        samples = zeros(shape=(num_samples,self.num_candidates,), dtype=float)
        for i in range(num_candidates):
            samples[:,i]  = bandit_est.generate_samples(num_samples, self.trials[i], self.wins[i])
    
        # calculate probability that each candidate is the best as a percentage
        winner = zeros(shape=(num_samples,self.num_candidates,), dtype=bool)
        for i, row in enumerate(samples):
            winner[i,argmax(row)] = True
        prob_winner = sum(winner,0)/num_samples
        leader_id = argmax(prob_winner)
        
        # calculate 95th percentile of remaining value distribution for further trials
        value_remaining = zeros(shape=(num_samples,), dtype=float)
        for i, row in enumerate(samples):
            value_remaining[i] = max(row) - row[leader_id]
        bins = linspace(0,1.0,999)  # even 1/1000 spacing
        hist, bins = histogram(value_remaining, bins=bins)
        idx = 0
        cum_value = cumsum(hist)/num_samples
        for i in range(len(bins)):
            if cum_value[i] >= 0.95:
                idx = i
                break
                
        return prob_winner, bins[idx]
                       
class bandit(object):
    def __init__(self, p=0.01):
        self.p = p
       
    def trial(self):
        return random.random() < self.p

import sys

theta = [0.04, 0.05, 0.045]
candidates = [bandit(x) for x in theta]
num_candidates = len(candidates)

bandit_est = bandit_estimator_betadist(num_candidates, (1.0,1.0))

num_trials = 100000
cum_trials_hist = zeros(shape=(num_trials,num_candidates), dtype=int)
cum_wins_hist = zeros(shape=(num_trials,num_candidates), dtype=int)
leader_id = zeros(1)
prob_winner = zeros(shape=(num_candidates,))
value_left = zeros(1)

for i in range(num_trials):
    # Pick the next trial candidate, do the trial, store the result
    candidate_id = bandit_est.pick_candidate_for_trial()
    result = candidates[candidate_id].trial()

    # Update distribution of trial candidate
    bandit_est.update_on_trial(candidate_id, result)

    # Store historical values for graphing 
    cum_trials_hist[i] = bandit_est.trials
    cum_wins_hist[i] = bandit_est.wins
    
    if (i+1) % 1000 == 0:
        # Every thousand loops, evaluate progress
        prob_winner, value_left = bandit_est.evaluate_progress(100000)
        leader_id = argmax(prob_winner)
        
        # Output progress indicator
        sys.stdout.write('\r' + str(i+1) + ' trials done, probabilities {}, 95th percentile of value remaining is {}'.format(prob_winner,
                                                                                                                             value_left))
        sys.stdout.flush()
        if (max(prob_winner) > 0.95) or (value_left <= 0.01*bandit_est.candidate_mode(leader_id)):
            break
        
print('\nTrials complete'.format(i+1))
print('Candidate {} wins with {}% conf., there is >95% confidence that <1% relative improvement remains.'.format(leader_id+1, 
                                                                                                         prob_winner[leader_id]*100, 
                                                                                                         value_left,))
num_trials = i+1
cum_trials_hist = cum_trials_hist[:num_trials]
cum_wins_hist = cum_wins_hist[:num_trials]
